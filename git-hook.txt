#!/bin/sh
# gogs响应客户端的推送，并通过本地的my-repo推送到远端web1-repo

#判断是不是远端仓库

IS_BARE=$(git rev-parse --is-bare-repository)
if [ -z "$IS_BARE" ]; then
echo >&2 "fatal: post-receive: IS_NOT_BARE"
exit 1
fi

unset GIT_DIR

echo "==============================================="
/usr/bin/whoami

ssh gtl@35.203.183.38 '/bin/bash /home/gtl/local.sh'
ssh root@35.203.183.38 '/bin/bash /data/wwwroot/local.sh'

echo "====================done======================="
